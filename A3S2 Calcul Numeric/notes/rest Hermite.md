$$
\begin{aligned}
(R_nf)(x)&=\frac{u(x)}{(n+1)!}f^{(n+1)}(\xi) \\
u(x)&=(x-x_0)^{r_0+1}(x-x_1)^{r_1+1}\dots(x-x_m)^{r_m+1} \\
\xi&\in(\min{\{x_0,\dots,x_m\}},\max{\{x_0,\dots,x_m\}}) \\
\end{aligned}
$$
$$
n+1=m+1+r_0+\dots+r_m
$$
